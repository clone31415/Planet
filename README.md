# Planet simulation

rel pos = (x, y)

magnitude = r

d = r^2 = x^2 + y^2

normalise (x, y) = (x / r, y / r)

---

ma = GMmr^-2

a = GMr^-2

---

a(x) = GMr^-2 * x / r

a(x) = GMxr^-3

a(x) = GMx * sqrt(d)^-3

a(x) = GMxd^-1.5

---

a(y) = GMr^-2 * y / r

a(y) = GMyr^-3

a(y) = GMy * sqrt(d)^-3

a(y) = GMyd^-1.5
